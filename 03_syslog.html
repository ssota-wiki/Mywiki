SYSLOG（rsyslog）

Rev 1.02026-01-15Author: CTCSP YO
// Table of Contents
  1. rsyslog インストール
  2. rsyslog 設定
  3. 動作テスト
  4. logrotate 設定

1. rsyslog インストール

bash
dnf -y update && dnf -y install rsyslog

バックアップ & ディレクトリ作成

bash
cp -p /etc/rsyslog.conf /etc/rsyslog.conf.org \
&& mkdir /var/log/syslog \
&& chmod 755 /var/log/syslog

2. rsyslog 設定

変数を設定

bash
SYSLOG_FACILITY='local7'

rsyslog.conf グローバル設定

bash
sed -i 's/^global(workDirectory/#&/' /etc/rsyslog.conf
sed -i '/^#global(workDirectory=/a\
global(\n    workDirectory="/var/lib/rsyslog"\n    net.enableDNS="on"\n)' /etc/rsyslog.conf

モジュール設定（UDP 514番受信）

bash
cat << 'EOF' > /etc/rsyslog.d/00-module.conf
### ListenPort of syslog
module(load="imudp") # needs to be done just once
input(type="imudp" port="514")

### Use UnixSocket (omuxsock)
module(load="omuxsock")
EOF

WARNING ログ設定

bash
cat << EOF > /etc/rsyslog.d/10-warning.conf
*.warn          /var/log/warning.log
&               stop
EOF

ネットワーク SYSLOG 設定（ホスト別ファイル振り分け）

bash
cat << EOF > /etc/rsyslog.d/21-netowork.conf
# Make Template of file
template(name="DynamicSYSLOG" type="list") {
    constant(value="/var/log/syslog/")
    property(name="fromhost")
    constant(value=".log")
}

# facility & output of file
${SYSLOG_FACILITY}.* ?DynamicSYSLOG
&                         stop
EOF

設定チェック & サービス再起動

bash
rsyslogd -N1
systemctl restart rsyslog.service
lsof -i:514

3. 動作テスト

bash
logger -p user.warn     -t rsyslogd "TEST-01: Testing SYSLOG of warning"
logger -p ${SYSLOG_FACILITY}.notice -t rsyslogd "TEST-02: Testing SYSLOG of network"

grep -E "Testing" /var/log/warning.log
grep -E "Testing" /var/log/syslog/*.log

4. logrotate 設定

変数を設定

bash
LOGROTATE_FREQUENCY='daily'
LOGROTATE_GENERATION='7'

バックアップ

bash
cp -p /etc/logrotate.conf /etc/logrotate.conf.org
cp -pr /etc/logrotate.d /etc/logrotate.d.org
mkdir /var/log/old

logrotate.conf デフォルト設定変更

bash
sed -i '/^weekly/s/^/#/' /etc/logrotate.conf
sed -i '/^rotate /s/^/#/' /etc/logrotate.conf
sed -i "/^#weekly/a ${LOGROTATE_FREQUENCY}" /etc/logrotate.conf
sed -i "/^#rotate /a rotate ${LOGROTATE_GENERATION}" /etc/logrotate.conf
sed -i "/^rotate ${LOGROTATE_GENERATION}/a notifempty" /etc/logrotate.conf
sed -i '/^#dateext/a dateext\ndateyesterday' /etc/logrotate.conf
sed -i '/^#compress/a compress\ncompresscmd /usr/bin/xz\nuncompresscmd /usr/bin/unxz\ncompressext .xz\ncompressoptions -3' /etc/logrotate.conf
sed -i '/^compressoptions -3/a \\nolddir /var/log/old\ncreateolddir 755 root root' /etc/logrotate.conf

rsyslog 用 logrotate 設定

bash
cat << 'EOF' > /etc/logrotate.d/rsyslog
/var/log/cron
/var/log/messages
/var/log/secure
/var/log/warning.log
/var/log/network.log
{
    missingok
    sharedscripts
    postrotate
        /usr/bin/systemctl kill -s HUP rsyslog.service >/dev/null 2>&1 || true
    endscript
}
EOF

テスト

bash
# dry-run（実際にはローテートしない）
logrotate -d /etc/logrotate.conf

# 強制ローテート
logrotate --verbose --force /etc/logrotate.conf

# 最終実行時刻の確認
cat /var/lib/logrotate/logrotate.status
