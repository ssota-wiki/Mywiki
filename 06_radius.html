0. Requirements

ItemSpecificationDescriptions
Server OSAlmaLinux 9 or later–
Server CPU2–4 core–
Server Mem2–4 GB–
Server Disk16–32 GB–
Internet Accessavailablesupports FQDN
Userroot–
Auth MethodXAuth or EAPEAP の場合は証明書ファイルが必要
⚠ SELinux — SELinux を無効化してください。手順は 01. AlmaLinux 9 を参照。

1. Install & Backup

パッケージのインストール

bash
dnf -y update \
  && dnf -y install lsof freeradius freeradius-utils \
  && dnf install -y openssl libnl3 dbus-libs

設定ファイルのバックアップ

bash
cp -p /etc/raddb/clients.conf /etc/raddb/clients.conf.org
cp -p /etc/raddb/radiusd.conf /etc/raddb/radiusd.conf.org
cp -p /etc/raddb/users /etc/raddb/users.org
cp -pr /etc/raddb/certs /etc/raddb/certs.org

XAuth

0. Requirements for Auth

ItemSpecificationDescriptions
Auth MethodXAuth–

1. 変数を設定

bash
RAD_CLIENT_NAME_1='auth_switch_1'
RAD_CLIENT_SUBNET_1='0.0.0.0/0'
RAD_CLIENT_SECRET_1='testing123'

2. 証明書の作成

bash
cd /etc/raddb/certs && ./bootstrap

3. clients.conf の設定

bash
cat <<EOF> /etc/raddb/clients.conf
# Local Test
client localhost {
        ipaddr          = 127.0.0.1
        proto           = *
        secret          = testing123
        nas_type        = other
        limit {
                max_connections = 16
                lifetime        = 0
                idle_timeout    = 30
        }
}
 
# Customs Setting for NAS
client ${RAD_CLIENT_NAME_1} {
        ipaddr          = ${RAD_CLIENT_SUBNET_1}
        secret          = ${RAD_CLIENT_SECRET_1}
}
EOF

4. radiusd.conf の設定（認証ログ有効化）

bash
sed -i 's/auth = no$/auth = yes/' /etc/raddb/radiusd.conf
sed -i 's/auth_badpass = no$/auth_badpass = yes/' /etc/raddb/radiusd.conf
sed -i 's/auth_goodpass = no$/auth_goodpass = yes/' /etc/raddb/radiusd.conf
 
diff /etc/raddb/radiusd.conf /etc/raddb/radiusd.conf.org

5. users の設定

bash
cat <<EOF> /etc/raddb/users
#################################################################
# XAuth (ID/PASS)
#
${RAD_USER_1} Cleartext-Password := "${RAD_PASS_1}"
#################################################################
# MAC-Auth (コメントアウトを外して使用)
#
#${RAD_MAC_USER_1} Cleartext-Password := "${RAD_MAC_PASS_1}"
#        Tunnel-Type             = "VLAN" ,
#        Tunnel-Medium-Type      = "IEEE-802" ,
#        Tunnel-Private-Group-Id = "${RAD_MAC_VLAN_1}"
#################################################################
DEFAULT Framed-Protocol == PPP
        Framed-Protocol = PPP,
        Framed-Compression = Van-Jacobson-TCP-IP
DEFAULT Hint == "CSLIP"
        Framed-Protocol = SLIP,
        Framed-Compression = Van-Jacobson-TCP-IP
DEFAULT Hint == "SLIP"
        Framed-Protocol = SLIP
EOF

1. 変数を設定

bash
# Radius
RAD_CLIENT_NAME_1='auth_switch_1'
RAD_CLIENT_SUBNET_1='0.0.0.0/0'
RAD_CLIENT_SECRET_1='testing123'
RAD_USER_1='ctcsp'
RAD_PASS_1='ctcsp'
RAD_MAC_VLAN_1='authvlan'
 
# 証明書の有効期限（日数）
CRT_EXPIRE='3650'
CRT_URI='dev-radius.sp.internal'
CRT_ON='JapanEast SE'
CRT_COUNTRY='JP'
CRT_STATE='Tokyo'
CRT_CITY='Minato-ku'
CRT_KEY_STRINGS='ctcsp'
 
# メールアドレス
CRT_CA_EMAIL='[email protected]'
CRT_SV_EMAIL='[email protected]'
CRT_CL_EMAIL='[email protected]'
 
# Common Name
CRT_CA_COMMON='Dev.Radius CA'
CRT_SV_COMMON='dev-radius.sp.internal'
CRT_CL_COMMON='[email protected]'
 
# SAN（ワイルドカード）
CRT_SV_OTHE_NAME='*.sp.internal'

2. 証明書の作成

*.cnf の設定

bash
cd /etc/raddb/certs
 
sed -i "s/\(default_days[[:space:]]*=[[:space:]]*\).*/\1${CRT_EXPIRE}/" *.cnf
sed -i "s/\(default_crl_days[[:space:]]*=[[:space:]]*\).*/\1${CRT_EXPIRE}/" *.cnf
sed -i "s|\(crlDistributionPoints[[:space:]]*=[[:space:]]*\).*|\1URI:http://${CRT_URI}/ca.crl|" *.cnf
sed -i "s/\(countryName[[:space:]]*=[[:space:]]*\)FR/\1${CRT_COUNTRY}/" *.cnf
sed -i "s/\(stateOrProvinceName[[:space:]]*=[[:space:]]*\)Radius/\1${CRT_STATE}/" *.cnf
sed -i "s/\(localityName[[:space:]]*=[[:space:]]*\)Somewhere/\1${CRT_CITY}/" *.cnf
sed -i "s/\(organizationName[[:space:]]*=[[:space:]]*\)Example Inc./\1${CRT_ON}/" *.cnf
sed -i "s/whatever/${CRT_KEY_STRINGS}/" passwords.mk
sed -i "s/[email protected]/${CRT_CL_EMAIL}/" passwords.mk
sed -i "s/\(emailAddress[[:space:]]*=[[:space:]]*\)[email protected]/\1${CRT_CA_EMAIL}/" ca.cnf
sed -i "s/\(emailAddress[[:space:]]*=[[:space:]]*\)[email protected]/\1${CRT_SV_EMAIL}/" server.cnf
sed -i "s/\(emailAddress[[:space:]]*=[[:space:]]*\)[email protected]/\1${CRT_CL_EMAIL}/" client.cnf
sed -i "s/\(commonName[[:space:]]*=[[:space:]]*\)\"Example Certificate Authority\"/\1\"${CRT_CA_COMMON}\"/" ca.cnf
sed -i "s/\(commonName[[:space:]]*=[[:space:]]*\)\"Example Server Certificate\"/\1\"${CRT_SV_COMMON}\"/" server.cnf
sed -i "s/\(commonName[[:space:]]*=[[:space:]]*\)[email protected]/\1${CRT_CL_COMMON}/" client.cnf
sed -i "s/\(otherName.0[[:space:]]*=[[:space:]]*\).*/\11.3.6.1.5.5.7.8.8;FORMAT:UTF8,UTF8:${CRT_SV_OTHE_NAME}/" server.cnf

証明書ファイルの生成

bash
rm -f *.pem *.der *.p12 *.csr *.key *.crt *.old index.txt* serial*
make ca server USER_NAME="${CRT_CL_COMMON}" client
chown -R root:radiusd ./ && cd

4. Test

XAuth テスト

変数を設定

bash
RAD_USER='ctcsp'
RAD_PASS='ctcsp'
RAD_TYPE='pap'           # pap|chap|mschap|eap-md5
RAD_SERVER='localhost'
RAD_SECRET='testing123'

radtest

bash
radtest -t ${RAD_TYPE} ${RAD_USER} ${RAD_PASS} ${RAD_SERVER} 0 ${RAD_SECRET}
output – Success
Sent Access-Request Id 84 from 0.0.0.0:35243 to 127.0.0.1:1812 length 75
        User-Name = "ctcsp"
        User-Password = "ctcsp"
        NAS-IP-Address = 192.168.100.23
        NAS-Port = 0
Received Access-Accept Id 84 from 127.0.0.1:1812 to 127.0.0.1:35243 length 38
output – Failed
Received Access-Reject Id 45 from 127.0.0.1:1812 to 127.0.0.1:37752 length 38
(0) -: Expected Access-Accept got Access-Reject

radclient

bash
echo "User-Name = '${RAD_USER}', User-Password = '${RAD_PASS}', NAS-IP-Address = 127.0.0.1" | \
radclient -s -x ${RAD_SERVER}:1812 auth ${RAD_SECRET}

EAP（MS-PEAP）テスト

変数を設定

bash
RAD_USER='ctcsp'
RAD_PASS='ctcsp'
RAD_CONF='test_ms-peap.conf'
RAD_SERVER='127.0.0.1'
RAD_SECRET='testing123'

テスト設定ファイルを作成

bash
cat <<EOF> ${RAD_CONF}
network={
    key_mgmt=WPA-EAP
    eap=PEAP
    identity="${RAD_USER}"
    password="${RAD_PASS}"
    phase2="auth=MSCHAPV2"
    ca_cert="/etc/raddb/certs/ca.pem"
}
EOF

認証テスト実行

bash
eapol_test -c ${RAD_CONF} -a ${RAD_SERVER} -s ${RAD_SECRET}
output – 最終行（Success）
MPPE keys OK: 1  mismatch: 0
SUCCESS

ログで認証結果を確認

bash
grep -E "eap" /var/log/radius/radius.log
output

Mon Jan 12 18:43:48 2026 : Auth: (25) Login OK: [ctcsp/<via Auth-Type = eap>] (from client localhost port 0 cli 02-00-00-00-00-01)

5. Trouble Shooting & Debugging

デバッグモードで起動（エラー確認）

bash
# "Ready to process requests" が出れば正常
systemctl stop radiusd && radiusd -X
output（正常時の末尾）
Listening on auth address * port 1812 bound to server default
Listening on acct address * port 1813 bound to server default
Listening on auth address 127.0.0.1 port 18120 bound to server inner-tunnel
Listening on proxy address * port 48431
Ready to process requests

認証ログの確認

bash
# 認証結果サマリー（RAD_TYPE に pap, eap などを指定）
grep -E "${RAD_TYPE}" /var/log/radius/radius.log
 
# 認証結果詳細（開始セッション）
awk 'BEGIN {RS=""} /Nov 21 17:03/ && /Acct-Status-Type = Start/' /var/log/radius/radacct/*/*
