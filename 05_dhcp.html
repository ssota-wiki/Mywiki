ℹ 構成 — Kea DHCP + MariaDB でリース情報を管理します。DHCP-Relay 経由の複数サブネット対応です。

1. Kea & MariaDB インストール

リポジトリ追加（Kea）

bash
cat <

リポジトリ追加（MariaDB）

bash
cat <<'EOF' >/etc/yum.repos.d/MariaDB.repo
[mariadb]
name = MariaDB
baseurl = https://ftp.yz.yamagata-u.ac.jp/pub/dbms/mariadb/yum/11.4/rhel/$releasever/$basearch
gpgkey = https://ftp.yz.yamagata-u.ac.jp/pub/dbms/mariadb/yum/RPM-GPG-KEY-MariaDB
gpgcheck = 1
EOF

パッケージのインストール

bash
dnf -y install isc-kea MariaDB-server MariaDB-client

バックアップ & kea-dhcp4 を MariaDB 起動後に依存させる設定

bash
for f in /etc/kea/*.conf; do cp "$f" "$f.org"; done \
&& chown -R kea:kea /etc/kea/

mkdir -p /etc/systemd/system/kea-dhcp4.service.d/
cat < /etc/systemd/system/kea-dhcp4.service.d/dependency.conf
[Unit]
Requires=mariadb.service
After=mariadb.service
EOF

systemctl daemon-reload

2. DB 初期化

bash
systemctl enable --now mariadb

# 対話型の初期設定（パスワード設定・不要アカウント削除）
mariadb-secure-installation

3. DB セットアップ

変数を設定

bash
DB_USER="ctcsp"
DB_PASS="ctcsp"

DB チューニング & ローカルバインド

bash
sed -i '0,/^\[mariadb\]/ s/^\[mariadb\]/&\ninnodb_flush_log_at_trx_commit=2/' /etc/my.cnf.d/server.cnf
sed -i '/^\#bind-address/a bind-address=127.0.0.1' /etc/my.cnf.d/server.cnf
systemctl restart mariadb

DB・ユーザー作成 & Kea スキーマ初期化

bash
mariadb -u root -e "CREATE DATABASE dhcp;"
mariadb -u root -e "CREATE USER '${DB_USER}'@'localhost' IDENTIFIED BY '${DB_PASS}';"
mariadb -u root -e "GRANT ALL ON dhcp.* TO '${DB_USER}'@'localhost';"
mariadb -u root -e "FLUSH PRIVILEGES;"
kea-admin db-init mysql -u ${DB_USER} -p ${DB_PASS} -n dhcp

確認

bash
mariadb -u root -e "SHOW DATABASES LIKE 'dhcp';"
mariadb -u ${DB_USER} -p"${DB_PASS}" -D dhcp -e "show tables;"

4. DHCP 設定

変数を設定

bash
# Global
IFACE="eth3"
LOG_LEVEL="INFO"
DNS_SERVER="130.49.0.41"
DOMAIN_NAME="sp.internal"
LEASE_TIME=3600
HOLD_TIME=600

# Subnet 0: Local (via BroadCast)
SUBNET_NAME="LOCAL"
SUBNET_NET="130.49.0.0/16"
SUBNET_POOL="130.49.255.100 - 130.49.255.150"
SUBNET_GW="130.49.0.1"

# Subnet 1: Relay (via DHCP-Relay)
SUBNET1_NAME="GUEST-VLAN300"
SUBNET1_NET="130.29.0.0/16"
SUBNET1_POOL="130.29.0.2 - 130.29.255.249"
SUBNET1_GW="130.29.11.1"
SUBNET1_RELAY="130.29.11.1"

# Subnet 2: Relay (via DHCP-Relay)
SUBNET2_NAME="AUTH-VLAN600"
SUBNET2_NET="160.29.0.0/16"
SUBNET2_POOL="160.29.0.2 - 160.29.255.249"
SUBNET2_GW="160.29.11.1"
SUBNET2_RELAY="160.29.11.1"

設定ファイルの検証 & サービス起動

bash
kea-dhcp4 -t /etc/kea/kea-dhcp4.conf
systemctl enable --now kea-dhcp4
lsof -i:67

5. デバッグ & テスト

リース情報の確認

bash
mariadb -u ${DB_USER} -p${DB_PASS} -D dhcp -t \
  -e "SELECT INET_NTOA(address) AS IP, HEX(hwaddr) AS MAC, hostname AS HOST, expire AS EXPIRE FROM lease4;"
output (例)
+----------------+--------------+---------+---------------------+
| IP             | MAC          | HOST    | EXPIRE              |
+----------------+--------------+---------+---------------------+
| 130.29.0.2     | 00155DCDA702 | mng-201 | 2026-02-10 12:21:03 |
| 130.49.255.100 | 00155DCDA60E | svc-101 | 2026-02-10 12:11:13 |
| 160.29.0.2     | 00155DCDA702 | mng-201 | 2026-02-10 12:19:21 |
+----------------+--------------+---------+---------------------+

その他デバッグコマンド

bash
# ログ監視
tail /var/log/kea/dhcp4.log | grep -E "DHCP4_QUERY|DHCP4_LEASE_ALLOC"

# パケットキャプチャ
tcpdump -i ${IFACE} port 67 or port 68 -n -vv

# 特定IPのリース削除
mariadb -u ${DB_USER} -p${DB_PASS} -e "DELETE FROM dhcp.lease4 WHERE address = INET_ATON('160.29.0.2');"

# 全リース削除
mariadb -u ${DB_USER} -p${DB_PASS} -e "TRUNCATE TABLE dhcp.lease4;"
